ApiGen

Executes ApiGen command to generate documentation

<?php
// ApiGen Command
$this->taskApiGen('./vendor/apigen/apigen.phar')
     ->config('./apigen.neon')
     ->templateConfig('vendor/apigen/apigen/templates/bootstrap/config.neon')
     ->wipeout(true)
      ->run();
?>
